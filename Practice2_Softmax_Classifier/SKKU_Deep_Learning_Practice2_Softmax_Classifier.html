def naive_softmax_loss(Weights,X_data,Y_data,reg):
    """
     Inputs have D dimension, there are C classes, and we operate on minibatches of N examples.
    
     Inputs :
         - Weights : A numpy array of shape (D,C) containing weights.
         - X_data : A numpy array of shape (N,D) contatining a minibatch of data.
         - Y_data : A numpy array of shape (N,) containing training labels; 
               Y[i]=c means that X[i] has label c, where 0<=c<C.
         - reg : Regularization strength. (float)
         
     Returns :
         - loss as single float
         - gradient with respect to Weights; an array of sample shape as Weights
     """
    
    # Initialize the loss and gradient to zero
    softmax_loss = 0.0
    dWeights = np.zeros_like(Weights)
    
    ####################################################################################################
    # TODO : Compute the softmax loss and its gradient using explicit loops.                           # 
    #        Store the loss in loss and the gradient in dW.                                            #
    #        If you are not careful here, it is easy to run into numeric instability.                  #
    #        Don't forget the regularization.                                                          #
    #---------------------------------------WRITE YOUR CODE--------------------------------------------#
    data_length = X_tr.shape[0]
    for idx in range(data_length):
        Y_expected = np.array([])
        for W in Weights.T:
            Y_expected = np.append(Y_expected, np.exp(np.dot(W, X_tr[idx])))
        P = Y_expected/Y_expected.sum()
        softmax_loss = softmax_loss - np.log(P[Y_tr[idx]])
        
        P[Y_tr[idx]] = P[Y_tr[idx]] - 1
        for i in range(dWeights.shape[1]):
            dWeights[:,i] = dWeights[:,i] + X_tr[idx].T*P[i]
            
    softmax_loss = softmax_loss / data_length + reg*np.sum(Weights*Weights)
    dWeights = dWeights/data_length + reg*Weights/2
    #--------------------------------------END OF YOUR CODE--------------------------------------------#
    ####################################################################################################
    return softmax_loss, dWeights

def vectorized_softmax_loss(Weights, X_data, Y_data, reg):
    softmax_loss = 0.0
    dWeights = np.zeros_like(Weights)

    ####################################################################################################
    # TODO : Compute the softmax loss and its gradient using no explicit loops.                        # 
    #        Store the loss in loss and the gradient in dW.                                            #
    #        If you are not careful here, it is easy to run into numeric instability.                  #
    #        Don't forget the regularization.                                                          #
    #---------------------------------------WRITE YOUR CODE--------------------------------------------#
    data_length = X_data.shape[0]
    prob = np.exp(np.dot(X_data, Weights))
    prob = prob / np.sum(prob, axis=1).reshape(-1,1)
    softmax_loss = np.mean(-np.log(prob[np.arange(data_length), Y_data])) + reg*np.sum(Weights*Weights)
    
    prob_onehot = np.zeros_like(prob)
    prob_onehot[np.arange(data_length),Y_data] = 1
    dWeights = np.einsum('ij,ik->kj', X_data, prob-prob_onehot).T/data_length + reg*Weights/2
    #--------------------------------------END OF YOUR CODE--------------------------------------------#
    ####################################################################################################
    
    return softmax_loss, dWeights

Compare two versions. The two versions should compute the same results, but the vectorized version should be much faster.
