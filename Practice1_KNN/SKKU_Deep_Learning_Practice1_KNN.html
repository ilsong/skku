def getEuclideanDist(X_tr, X_te_instance):
    """
    getEuclideanDist  .
     X_te_instace, Training dataset(X_tr)  euclidean distance  
    
    Inputs : 
        - X_tr : (N, D)  ( Training data)
        - X_te_instance : (1, D)   ( Test data)
    
    Output : 
        - dist_arr : N distance
    
    Hint :
        - Numpy linalg.norm()   .
        
    """
    
    dist_arr = []
    length = X_tr.shape[0]
    
    #########################################################################################################
    #------------------------------------------WRITE YOUR CODE----------------------------------------------#
    dist_arr = np.linalg.norm(X_tr - X_te_instance, axis=1, ord=2)
    #-----------------------------------------END OF YOUR CODE----------------------------------------------#
    #########################################################################################################
    
    return dist_arr

def getKNN(distances, Y_tr, k):
    """
    getKNN  .
    N distance    k Y_tr(Training data Label)  .
    
    Inputs : 
        - distances : N distance
        - Y_tr : (N, 1)   ( Training data Label)
        - k : KNN  k
    
    Output : 
        - neighbor :   k Training data Label
        
    Hint :
        - Numpy argpartition(), take()   .
        
    """
    
    neighbors = None
    
    #########################################################################################################
    #------------------------------------------WRITE YOUR CODE----------------------------------------------#
    idx = np.argpartition(distances, k)
    return np.take(Y_tr, idx[:k])
    #-----------------------------------------END OF YOUR CODE----------------------------------------------#
    #########################################################################################################
    
    return neighbors

def getPredictedLabel(neighbors):
    """
    getPredictedLabel  .
    k   training data label  Predicted label  .
    k neighbors     predicted label .
    
    Inputs : 
        - neighbors :   k Training data Label
    
    Output : 
        - predicted_label : neighbor   1 predicted label
    
    Hint : 
        - Numpy bincount()   .
        
    """
    
    predicted_label = None
    
    #########################################################################################################
    #------------------------------------------WRITE YOUR CODE----------------------------------------------#
    predicted_label = np.argmax(np.bincount(neighbors))
    #-----------------------------------------END OF YOUR CODE----------------------------------------------#
    #########################################################################################################
    
    return predicted_label

def getKNNAccuray(k,X_tr,Y_tr,X_te,Y_te):
    """
    getKNNAccuray  .
    training data   test data KNN  accuracy  .
       getEuclideanDist(), getKNN(), getPredictedLabel()   .
    for   test data   predicted label .
    
    Inputs : 
        - k : KNN  k
        - X_tr : training dataset input data
        - Y_tr : training dataset label data
        - X_te : test dataset input data
        - Y_te : test dataset label data
        
    Output : 
        - accuracy : KNN    test dataset accuracy
        
    """
    
    predicted_label_arr = []
    accuracy = None
    
    length = X_te.shape[0]
    
    #########################################################################################################
    #------------------------------------------WRITE YOUR CODE----------------------------------------------#
    for X in X_te:
        distances = getEuclideanDist(X_tr, X)
        neighbors = getKNN(distances, Y_tr, k)
        predicted_label_arr.append(getPredictedLabel(neighbors))
    accuracy = (predicted_label_arr == Y_te).sum() / Y_te.shape[0]
    #-----------------------------------------END OF YOUR CODE----------------------------------------------#
    #########################################################################################################
    
    return accuracy
