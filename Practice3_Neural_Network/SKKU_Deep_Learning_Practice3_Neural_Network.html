def softmax(x):
    
    """
    Softmax  .
    NaN     ( )
    
    Inputs : 
        - x : (N,D)  
    
    Output : 
        - softmax_output : (N,D)  Softmax 
    
    """
    
    softmax_output = None
    
    #########################################################################################################
    #------------------------------------------WRITE YOUR CODE----------------------------------------------#
    softmax_exp = np.exp(x-np.min(x, axis=1).reshape(-1,1))
    softmax_output = softmax_exp / np.sum(softmax_exp, axis=1).reshape(-1,1)
    #-----------------------------------------END OF YOUR CODE----------------------------------------------#
    #########################################################################################################
    
    return softmax_output

def cross_entropy_loss(softmax_output, target, weights, regularization):
    
    """
    cross_entropy_loss  .
    delta  log input 0   
    Weights  L2 Regularization  
    L2 Regularization 0.5   (    )
    
    Inputs : 
        - softmax_output : Softmax   (N, D)  
        - target : (N, # of labels)   (One-hot encoding)
        - weights : NN    Weight (Dictionary )
        - regularization : regularization  0 1 
    
    Output : 
        - loss : scalar loss 
    """
    
    delta = 1e-9
    batch_size = target.shape[0]
    data_loss = 0
    reg_loss = 0
    loss = None
    
    #########################################################################################################
    #------------------------------------------WRITE YOUR CODE----------------------------------------------#
    weights_list = list(weights.values())
    data_loss = np.mean(-np.log(softmax_output[range(len(softmax_output)), np.argmax(target, axis=1)]+delta))
    for weight in weights_list:
        reg_loss = reg_loss + regularization*np.sum(weight**2)/2
    #-----------------------------------------END OF YOUR CODE----------------------------------------------#
    #########################################################################################################
    
    loss = data_loss + reg_loss
    
    return loss

class OutputLayer:
    
    """
    Softmax  Cross-entropy loss  Ouput Layer class .
      softmax()  cross_entropy_loss    
    forward, backward    
    backpropagation softmax output target label     
    
    forward() : 
        - x : Forward propagation     (N,D)  
        - y : (N, # of Label)   
        - return : softmax loss
    
    backward() : 
        - dout : backpropagation      delta , output layer  delta = 1
        - return : dx

    """
    
    def __init__(self, weights, regularization):
        self.loss = None           # loss value
        self.output_softmax = None # Output of softmax
        self.target_label = None   # Target label (one-hot vector)
        self.weights = weights
        self.regularization = regularization
        
    def forward(self, x, y):
    
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        self.output_softmax = softmax(x)
        self.target_label = y
        self.loss = cross_entropy_loss(self.output_softmax, self.target_label, self.weights, self.regularization)
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################
    
        return self.loss
    
    def backward(self, dout=1):
        
        bt_size = self.target_label.shape[0]
        dx = None
        
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        dx = (self.output_softmax-self.target_label*dout) / bt_size
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################
        
        return dx

class ReLU:
    
    """
    ReLU 
    forward, backward    
    backpropagation forward    mask     
    
    forward() : 
        - x : Forward propagation     (N,D)  
        - return : ReLU output
    
    backward() : 
        - dout : backpropagation      delta 
        - return : dx
    
    """
    
    def __init__(self):
        
        self.mask = None
        
    def forward(self, x):
        
        self.out = None
    
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        self.mask = x > 0
        self.out = x * self.mask
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################
    
        return self.out
    
    def backward(self, dout):
    
        dx = None
        
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        dx = dout * self.mask
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################
        
        return dx

class Sigmoid:
    
    """
    Sigmoid 
    forward, backward    
    backpropagation  forward        
    
    forward() : 
        - x : Forward propagation     (N,D)  
        - return : Sigmoid output
    
    backward() : 
        - dout : backpropagation      delta 
        - return : dx
    
    """
    
    def __init__(self):
        self.out = None
        
    def forward(self, x):
    
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        self.out = 1/(1+np.exp(-x))
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################
    
        return self.out
    
    def backward(self, dout):
        
        dx = None
        
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        dx = self.out*(1-self.out)*dout
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #############################################################################################s############
        
        return dx

class Affine:
    
    """
    Affine  
    Affine  Y = np.dot(X, W) + B (X , W , B )      
    forward, backward    
    Backward  W,b  x      
    
    forward() : 
        - x : Forward propagation     (N,D)  
        - return : Affine output
    
    backward() : 
        - dout : backpropagation      delta 
        - return : dx
    
    """
    
    def __init__(self, W, b):
        self.W = W
        self.b = b
        self.x = None
        self.dW = None
        self.db = None
        
    def forward(self, x):
        
        out = None
    
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        self.x = x
        out = self.x.dot(self.W) + self.b
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################
    
        return out
    
    def backward(self, dout):
        
        dx = None
        
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        dx = dout.dot(self.W.T)
        self.dW = dout.T.dot(self.x).T
        self.db = np.sum(dout, axis=0)
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################
        
        return dx

from collections import OrderedDict
from utils import numerical_gradient
class TwoLayerNet:
    
    """

    __init__() : 
        - Weight  bias initialization  
        - Initialization  Weight bias    
    
    predict() : 
        - Input data(x)  Neural network forward propagtion  
        
    loss() : 
        - Input data(x)  Neural network forward propagtion    Loss  
        
    accuracy() :
        - Input data(x)  True label(y)  Accuracy  
        
    numerical_gradient() :
        - Input data(x) True label(y)  Numerical_gradient  
        - Backpropagation    
        
    gradient():
        - Input data(x) True label(y)  Backpropagation  

    """

    def __init__(self, input_size, hidden_size, output_size, weight_init_std = 0.01, regularization = 0.0):

        #  
        # Weight initialization
        self.params = {}
        self.params['W1'] = weight_init_std * np.random.randn(input_size, hidden_size)
        self.params['b1'] = np.zeros(hidden_size)
        self.params['W2'] = weight_init_std * np.random.randn(hidden_size, output_size) 
        self.params['b2'] = np.zeros(output_size)

        self.weights = {}
        self.weights['W1'] = self.params['W1']
        self.weights['W2'] = self.params['W2']
        
        self.reg = regularization

        #  
        # Layer generation
        self.layers = OrderedDict() # infromation about OrderedDict (https://pymotw.com/2/collections/ordereddict.html)
        
        #########################################################################################################
        # TwoLayerNet                                                                                  
        # Neural Network                                                                
        # [ Input => Fully Connected => ReLU => Fully Connected(OutputLayer) ]
        #     Class(Affine, ReLU)  Weight initialization   weight   
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        self.layers['Affine1'] = Affine(self.weights['W1'], self.params['b1'])
        self.layers['Relu1'] = ReLU()
        self.layers['Affine2'] = Affine(self.weights['W2'], self.params['b2'])
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################
        self.lastLayer = OutputLayer(self.weights, self.reg)

    def predict(self, x):

        for layer in self.layers.values():
            x = layer.forward(x)
            
        return x

    def loss(self, x, y):
        score = self.predict(x)
        return self.lastLayer.forward(score, y)

    

    def accuracy(self, x, y):

        score = self.predict(x)
        score = np.argmax(score, axis=1)
        if y.ndim != 1 : y = np.argmax(y, axis=1)
        accuracy = np.sum(score == y) / float(x.shape[0])

        return accuracy

    def numerical_gradient(self, x, y):

        loss_W = lambda W: self.loss(x, y)

        grads = {}
        grads['W1'] = numerical_gradient(loss_W, self.params['W1'])
        grads['b1'] = numerical_gradient(loss_W, self.params['b1'])
        grads['W2'] = numerical_gradient(loss_W, self.params['W2'])
        grads['b2'] = numerical_gradient(loss_W, self.params['b2'])
            
        return grads

        

    def gradient(self, x, y):

        # forward
        self.loss(x, y)

        # backward
        dout = 1
        dout = self.lastLayer.backward(dout)
        layers = list(self.layers.values())
        layers.reverse()

        for layer in layers:
            dout = layer.backward(dout)

        grads = {}
        grads['W1'], grads['b1'] = self.layers['Affine1'].dW, self.layers['Affine1'].db
        grads['W2'], grads['b2'] = self.layers['Affine2'].dW, self.layers['Affine2'].db
        
        if self.reg != 0.0:
            
            #########################################################################################################
            #  Weight Gradient   Regularization   
            #########################################################################################################
            #------------------------------------------WRITE YOUR CODE----------------------------------------------#
            grads['W1'] = grads['W1'] + self.reg*self.params['W1']
            grads['b1'] = grads['b1'] + self.reg*self.params['b1']
            grads['W2'] = grads['W2'] + self.reg*self.params['W2']
            grads['b2'] = grads['b2'] + self.reg*self.params['b2']
            #-----------------------------------------END OF YOUR CODE----------------------------------------------#
            #########################################################################################################
            
        return grads

class ThreeLayerNet:
    
    """

    __init__() : 
        - Weight  bias initialization  
        - Initialization  Weight bias    
    
    predict() : 
        - Input data(x)  Neural network forward propagtion  
        
    loss() : 
        - Input data(x)  Neural network forward propagtion    Loss  
        
    accuracy() :
        - Input data(x)  True label(y)  Accuracy  
        
    numerical_gradient() :
        - Input data(x) True label(y)  Numerical_gradient  
        - Backpropagation    
    gradient():
        - Input data(x) True label(y)  Backpropagation  
        
    """

    def __init__(self, input_size, hidden_size1, hidden_size2, output_size, weight_init_std = 0.01, regularization = 0.0):

        
        #########################################################################################################
        # TwoLayerNet  ThreeLayerNet 
        # Neural Network     
        #[ Input => Fully Connected => ReLU => Fully Connected => ReLU => Fully Connected(OutputLayer) ]
        #    Class   
        #* Hidden Layer    (Weight  bias , Hidden Layer weight , Weight update )
        #    *
        # Hidden Layer  hidden_size1, hidden_size2  
        
        
        # Implement ThreeLayerNet applying TwoLayerNet
        # Implement Neural Network structure as follow:
        #[ Input => Fully Connected => ReLU => Fully Connected => ReLU => Fully Connected(OutputLayer) ]
        # Implement using the Class previously made
        # * Consider the elements to be changed as Hidden Layer increase (e.g. addition of Weight and bias, number of weight in Hidden Layers, 
        #  Weight update) *
        # Use hidden_size1 and hidden_size2 as variables of Hidden Layer
        #########################################################################################################
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        self.params = {}
        self.params['W1'] = weight_init_std * np.random.randn(input_size, hidden_size1)
        self.params['b1'] = np.zeros(hidden_size1)
        self.params['W2'] = weight_init_std * np.random.randn(hidden_size1, hidden_size2) 
        self.params['b2'] = np.zeros(hidden_size2)
        self.params['W3'] = weight_init_std * np.random.randn(hidden_size2, output_size) 
        self.params['b3'] = np.zeros(output_size)

        self.weights = {}
        self.weights['W1'] = self.params['W1']
        self.weights['W2'] = self.params['W2']
        self.weights['W3'] = self.params['W3']
        
        self.reg = regularization
        self.layers = OrderedDict()
        
        self.layers['Affine1'] = Affine(self.weights['W1'], self.params['b1'])
        self.layers['Relu1'] = ReLU()
        self.layers['Affine2'] = Affine(self.weights['W2'], self.params['b2'])
        self.layers['Relu2'] = ReLU()
        self.layers['Affine3'] = Affine(self.weights['W3'], self.params['b3'])
        
        self.lastLayer = OutputLayer(self.weights, self.reg)
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################

    def predict(self, x):
        for layer in self.layers.values():
            x = layer.forward(x)
            
        return x

    def loss(self, x, y):        
        score = self.predict(x)
        
        return self.lastLayer.forward(score, y)

    

    def accuracy(self, x, y):
        score = self.predict(x)
        score = np.argmax(score, axis=1)
        if y.ndim != 1 : y = np.argmax(y, axis=1)
        accuracy = np.sum(score == y) / float(x.shape[0])

        return accuracy

        

    def gradient(self, x, y):
        # forward
        self.loss(x, y)
        # backward
        dout = 1
        dout = self.lastLayer.backward(dout)
        layers = list(self.layers.values())
        layers.reverse()

        for layer in layers:
            dout = layer.backward(dout)

        grads = {}
        #########################################################################################################
        #  Weight  bias Gradient   
        # Implement the part obtaining Gradient of each Weight and bias
        #########################################################################################################
        #------------------------------------------WRITE YOUR CODE----------------------------------------------#
        grads['W1'], grads['b1'] = self.layers['Affine1'].dW, self.layers['Affine1'].db
        grads['W2'], grads['b2'] = self.layers['Affine2'].dW, self.layers['Affine2'].db
        grads['W3'], grads['b3'] = self.layers['Affine3'].dW, self.layers['Affine3'].db
        
        if self.reg != 0.0:
            grads['W1'] = grads['W1'] + self.reg*self.params['W1']
            grads['b1'] = grads['b1'] + self.reg*self.params['b1']
            grads['W2'] = grads['W2'] + self.reg*self.params['W2']
            grads['b2'] = grads['b2'] + self.reg*self.params['b2']
            grads['W3'] = grads['W3'] + self.reg*self.params['W3']
            grads['b3'] = grads['b3'] + self.reg*self.params['b3']
        #-----------------------------------------END OF YOUR CODE----------------------------------------------#
        #########################################################################################################

        return grads
